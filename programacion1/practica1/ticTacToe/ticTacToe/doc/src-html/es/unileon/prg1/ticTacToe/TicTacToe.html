001package es.unileon.prg1.ticTacToe;
002import java.awt.GridLayout;
003import java.awt.event.ActionEvent;
004import java.awt.event.ActionListener;
005
006import javax.swing.ImageIcon;
007import javax.swing.JFrame;
008import javax.swing.JMenuBar;
009import javax.swing.JMenuItem;
010import javax.swing.JOptionPane;
011
012
013
014public class TicTacToe implements ActionListener {
015        
016        private JFrame window;
017        
018        private GameBoard gameBoard;
019        
020        private Player player1;
021
022        private Player player2;
023        
024        private JMenuBar menu;
025
026        private JMenuItem newGame;
027
028        private JMenuItem name;
029
030        private JMenuItem instr;
031
032        private JMenuItem exit;
033
034        public TicTacToe() {
035                // Create Window
036                this.window = new JFrame("Tic-Tac-Toe");
037                this.window.setSize(370, 370);
038                this.window.setLocationRelativeTo(null);
039                this.window.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
040                this.window.setLayout(new GridLayout(3, 3));
041
042                // Add Buttons To The Window
043                this.gameBoard = new GameBoard(this.window, this);
044
045                this.newGame = new JMenuItem("New Game");
046                this.name = new JMenuItem("Change Name");
047                this.instr = new JMenuItem("Instructions");
048                this.exit = new JMenuItem("Exit");
049                
050                this.menu = new JMenuBar();
051                
052                this.menu.add(this.newGame);
053                this.menu.add(this.name);
054                this.menu.add(this.instr);
055                this.menu.add(this.exit);
056
057                this.newGame.addActionListener(this);
058                this.name.addActionListener(this);
059                this.instr.addActionListener(this);
060                this.exit.addActionListener(this);
061                
062                this.window.setJMenuBar(this.menu);
063        }
064        
065        public void play(){
066                // Make The Window Visible
067                this.window.setVisible(true);
068                // Create the players only the first time
069                if ( (this.player1 == null) && (this.player2 == null) ) {
070                        this.setPlayers();
071                } else {
072                        this.gameBoard.setPlayers(player1, player2);
073                }
074        }
075        
076        public void setPlayers() {
077                String player1Name, player2Name;
078                
079                // JOptionPane belongs to javax.swing
080                player1Name = JOptionPane.
081                                showInputDialog(null, "Enter Name of player 1: ", "", 1);
082                player2Name = JOptionPane.
083                                showInputDialog(null, "Enter Name of player 2: ", "", 1);
084                // If the name introduced for the first player is null or empty
085                // then the name "1" is assigned to this player
086                if ( (player1Name == null) || (player1Name.length() == 0) ) {
087                        player1Name = "1";
088                }
089                // If the name introduced for the second player is null or empty
090                // then the name "2" is assigned to this player
091                if ( (player2Name == null) || (player2Name.length() == 0) ) {
092                        player2Name = "2";
093                }
094                JOptionPane
095                                .showMessageDialog(
096                                                null,
097                                                "Your names have been set\nTo change your names click on the Change name button in the menu bar",
098                                                "Name Changed!!!", JOptionPane.INFORMATION_MESSAGE);
099
100                this.player1 = new Player(
101                                player1Name, 
102                                new Mark(
103                                                new ImageIcon("etc/images/chrome_100x100.jpg")
104                                                ) 
105                                );
106                this.player2 = new Player(
107                                player2Name, 
108                                new Mark(
109                                                new ImageIcon("etc/images/firefox_100x100.jpg")
110                                                )
111                                );
112                
113                this.gameBoard.setPlayers(player1, player2);
114        }
115
116        public void actionPerformed(ActionEvent actionEvent) {
117                Object source = actionEvent.getSource();
118
119                if (source == this.newGame ) {
120                        int answer = JOptionPane
121                                        .showConfirmDialog(
122                                                        null,
123                                                        "Your current game will not be saved...\nContinue Anyways??",
124                                                        "Do you want to start a new game?",
125                                                        JOptionPane.YES_NO_OPTION);
126
127                        if (answer == JOptionPane.YES_OPTION) {
128                                this.gameBoard.clearIt();
129                        }
130
131                } else if (source == this.name) {
132                        this.setPlayers();
133
134                } else if (source == this.instr) {
135                        JOptionPane
136                                        .showMessageDialog(
137                                                        null,
138                                                        "Your goal is to be the first player to get 3 X's or O's in a row. (horizontally, diagonally, or vertically)",
139                                                        "Instructions", JOptionPane.INFORMATION_MESSAGE);
140                } else if (source == this.exit) {
141                        int answer = JOptionPane.showConfirmDialog(null, "EXIT",
142                                        "Are You sure you want to exit??",
143                                        JOptionPane.YES_NO_OPTION);
144
145                        if (answer == JOptionPane.YES_OPTION) {
146                                JOptionPane.showMessageDialog(null, "Thank you " + this.player1 + " and "
147                                                + this.player2 + " for playing");
148                                System.exit(0);
149                        }
150                }
151        }
152
153}
